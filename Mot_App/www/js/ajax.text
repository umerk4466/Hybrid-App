$("#login_form").submit(function(event) {
  event.preventDefault();
  $.ajax({
    url:
      "https://motproject01.pythonanywhere.com/api/get/vehicle/?vehicle_id=22",
    dataType: "json",
    type: "GET",
    timeout: 25000,
    success: function(data, status) {
      alert(data.message);
    },
    error: function() {
      $("#text").html("error");
    }
  });
  // window.location = "home.html";
});
